SELECT A.address, A.last_name as Surname_1, A.first_name as Name_1, B.last_name as Surname_2, B.first_name as Name_2 FROM Customers A, Customers B 
WHERE A.id <> B.id and A.address = B.address 
ORDER BY A.address;
